This is the docker definition for the Valence circuit & controller builder.

docker run --rm -i \
  -v $HOME/dev/timewave/valence-coprocessor-app:/mnt \

  domain \
  v0.1.13 \
  valence-coprocessor-app-domain \
  /mnt/crates/domain \
  104.171.203.127:37281 \
